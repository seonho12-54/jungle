(() => { if (window.storyMonitor && window.storyMonitor.stop) { window.storyMonitor.stop(); } const CONFIG = { relayUrl: "http://127.0.0.1:3000/api/slack-relay", queryValues: ["yexnduo", "jae.lol__"], intervalMs: 60_000, waitAfterSearchMs: 3000, betweenUsersMs: 2000, debug: true }; const STATE_KEY_PREFIX = "story_monitor_state_v4"; let running = false; function log(...args) { if (CONFIG.debug) { console.log("[story-monitor]", ...args); } } function sleep(ms) { return new Promise(resolve => setTimeout(resolve, ms)); } function getStateKey(username) { return `${STATE_KEY_PREFIX}_${username}`; } function loadState(username) { try { return JSON.parse(localStorage.getItem(getStateKey(username)) || "{}"); } catch { return {}; } } function saveState(username, state) { localStorage.setItem(getStateKey(username), JSON.stringify(state)); } function clearState(username) { localStorage.removeItem(getStateKey(username)); } function clearAllState() { for (const username of CONFIG.queryValues) { clearState(username); } } async function sendSlack(msg) { const res = await fetch(CONFIG.relayUrl, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ text: msg }) }); let data = null; try { data = await res.json(); } catch { data = null; } if (!res.ok || !data?.ok) { throw new Error( `relay failed: status=${res.status}, body=${JSON.stringify(data)}` ); } log("Slack 전송 완료"); } function runSearch(username) { const input = document.querySelector("#sf_url"); const button = document.querySelector("#sf_submit"); if (!input || !button) { throw new Error("검색 UI(#sf_url 또는 #sf_submit)를 찾지 못함"); } input.focus(); input.value = ""; input.dispatchEvent(new Event("input", { bubbles: true })); input.value = username; input.dispatchEvent(new Event("input", { bubbles: true })); input.dispatchEvent(new Event("change", { bubbles: true })); button.click(); log("검색 실행:", username); } function readStoryCount() { return document.querySelectorAll(".ig-stories__card").length; } function readUsername(fallbackUsername) { return ( document.querySelector(".ig-stories__user__name")?.textContent?.trim() || fallbackUsername ); } async function checkUser(username) { log("체크 시작:", username); runSearch(username); await sleep(CONFIG.waitAfterSearchMs); const currentCount = readStoryCount(); const detectedUsername = readUsername(username); const state = loadState(username); const previousCount = Number.isFinite(state.count) ? state.count : null; log( `[${username}] 현재 스토리 개수:`, currentCount, "| 이전:", previousCount ); // 첫 실행 if (previousCount === null) { if (currentCount > 0) { const msg = `스토리 감지 (첫 실행) 계정: ${detectedUsername} 현재: ${currentCount}`; await sendSlack(msg); log("첫 실행 알림 전송:", username); } else { log("첫 실행: 스토리 없음:", username); } saveState(username, { count: currentCount, username: detectedUsername, initializedAt: new Date().toISOString(), lastCheckedAt: new Date().toISOString() }); return; } // 이후에는 개수 증가할 때만 알림 if (currentCount > previousCount) { const added = currentCount - previousCount; const msg = `스토리 업데이트 감지 계정: ${detectedUsername} 이전: ${previousCount} 현재: ${currentCount} 추가: ${added}`; await sendSlack(msg); log("증가 알림 전송:", username); } else { log("알림 없음:", username); } saveState(username, { count: currentCount, username: detectedUsername, initializedAt: state.initializedAt || new Date().toISOString(), lastCheckedAt: new Date().toISOString() }); } async function checkOnce() { log("전체 체크 시작"); for (const username of CONFIG.queryValues) { try { await checkUser(username); } catch (e) { console.error(`[story-monitor] ${username} 체크 중 에러:`, e); } await sleep(CONFIG.betweenUsersMs); } } async function loop() { while (running) { try { await checkOnce(); } catch (e) { console.error("[story-monitor] 에러:", e); } if (!running) break; log(`다음 체크까지 ${CONFIG.intervalMs / 1000}초 대기`); await sleep(CONFIG.intervalMs); } log("모니터 종료"); } window.storyMonitor = { start() { if (running) { log("이미 실행 중"); return; } running = true; loop(); }, stop() { running = false; log("중단 요청"); }, reset(username) { if (username) { clearState(username); log(`상태 초기화 완료: ${username}`); } else { clearAllState(); log("전체 상태 초기화 완료"); } }, state(username) { if (username) { return loadState(username); } const result = {}; for (const name of CONFIG.queryValues) { result[name] = loadState(name); } return result; }, checkOnce }; log("스토리 모니터 시작"); window.storyMonitor.start(); })();
